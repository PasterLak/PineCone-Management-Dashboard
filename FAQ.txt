./install_toolchain.sh

#------------------------------------------------------------------------------#
## HOW SETUP THE PROJECT
#------------------------------------------------------------------------------#
 1) run  ./install_toolchain.sh in the terminal
 2) move the project to user/bl602_iot_sdk/customer_app
    (the project will not compile if not in the sdk folder :/ )
 3) run  ./compile.sh in terminal for compile the project
 4) run ./flash.sh in terminal to upload the compiled files to pinecone and follow the instructions in the terminal



## COMPILE  -> run:   ./compile.sh
## FLASH  -> run and follow the instructons:   ./flash.sh
## REBUILD -> same as clean + compile

## CLEAN  (clean compiled files )-> run:   ./clean.sh

for perfomance test:

extern "C" {

#include <math.h>
}

#include "components/profiler.hpp"


  Profiler profiler;
  printf("Start\n");
  volatile double result = 0.0;

  profiler.start();

  for (long i = 1; i < 1000000; i++) {
    result += sqrt(static_cast<double>(i));
  }

  result += 1;

  profiler.end();
  printf("Completed %d iterations\n", 1000000);
  profiler.printTime();
  printf("Result: %f\n", result);
